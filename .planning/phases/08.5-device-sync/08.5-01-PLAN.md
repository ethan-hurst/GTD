   ```typescript
   export async function compactTombstones(maxAgeDays: number = 30): Promise<number> {
     const cutoff = new Date(Date.now() - maxAgeDays * 24 * 60 * 60 * 1000);
     let count = 0;
     // Compact items
     const oldItems = await db.items.filter(item => item.deleted === true && item.deletedAt && item.deletedAt < cutoff).toArray();
     if (oldItems.length > 0) {
       await db.items.bulkDelete(oldItems.map(i => i.id));
       count += oldItems.length;
     }
     // Compact events
     const oldEvents = await db.events.filter(e => e.deleted === true && e.deletedAt && e.deletedAt < cutoff).toArray();
     if (oldEvents.length > 0) {
       await db.events.bulkDelete(oldEvents.map(e => e.id));
       count += oldEvents.length;
     }
     // Compact contexts
     const oldContexts = await db.contexts.filter(c => (c as any).deleted === true && (c as any).deletedAt && (c as any).deletedAt < cutoff).toArray();
     if (oldContexts.length > 0) {
       await db.contexts.bulkDelete(oldContexts.map(c => c.id));
       count += oldContexts.length;
     }
     return count;
   }
   ```
