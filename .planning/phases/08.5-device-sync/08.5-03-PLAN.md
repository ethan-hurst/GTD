3. Create `netlify/functions/sync-push.mts`:
   ```typescript
   import { getStore } from '@netlify/blobs';
   import type { Context } from '@netlify/functions';

   export default async (req: Request, context: Context) => {
     // Only allow POST
     if (req.method !== 'POST') {
       return new Response(JSON.stringify({ error: 'Method not allowed' }), {
         status: 405,
         headers: { 'Content-Type': 'application/json' }
       });
     }

     try {
       const body = await req.json();
       const { deviceId, encryptedBlob } = body;

       // Validate required fields
       if (!deviceId || typeof deviceId !== 'string') {
         return new Response(JSON.stringify({ error: 'Missing or invalid deviceId' }), {
           status: 400,
           headers: { 'Content-Type': 'application/json' }
         });
       }

       if (!encryptedBlob || typeof encryptedBlob !== 'string') {
         return new Response(JSON.stringify({ error: 'Missing or invalid encryptedBlob' }), {
           status: 400,
           headers: { 'Content-Type': 'application/json' }
         });
       }

       // Enforce max payload size (10 MB)
       if (encryptedBlob.length > 10 * 1024 * 1024) {
         return new Response(JSON.stringify({ error: 'Payload too large (max 10MB)' }), {
           status: 413,
           headers: { 'Content-Type': 'application/json' }
         });
       }

       // Store encrypted blob in Netlify Blobs
       const store = getStore('sync-data');
       await store.set(deviceId, encryptedBlob);

       return new Response(JSON.stringify({
         success: true,
         timestamp: new Date().toISOString()
       }), {
         status: 200,
         headers: { 'Content-Type': 'application/json' }
       });
     } catch (error: any) {
       console.error('sync-push error:', error);
       return new Response(JSON.stringify({
         error: 'Internal server error'
       }), {
         status: 500,
         headers: { 'Content-Type': 'application/json' }
       });
     }
   };
   ```

Key design decisions:
- Device ID is the SHA-256 hash of the pairing code (sent from client)
- Encrypted blob is a JSON string containing salt, IV, and ciphertext (all base64)
- Server never sees plaintext - just stores/retrieves opaque encrypted strings
- 10 MB limit as a safety valve (typical GTD data < 1 MB)
- No authentication beyond knowing the device ID (security through encryption)

Create `netlify/functions/sync-pull.mts`:

```typescript
import { getStore } from '@netlify/blobs';
import type { Context } from '@netlify/functions';

export default async (req: Request, context: Context) => {
  // Only allow GET
  if (req.method !== 'GET') {
    return new Response(JSON.stringify({ error: 'Method not allowed' }), {
      status: 405,
      headers: { 'Content-Type': 'application/json' }
    });
  }

  try {
    const url = new URL(req.url);
    const deviceId = url.searchParams.get('deviceId');

    // Validate deviceId
    if (!deviceId || typeof deviceId !== 'string') {
      return new Response(JSON.stringify({ error: 'Missing or invalid deviceId parameter' }), {
        status: 400,
        headers: { 'Content-Type': 'application/json' }
      });
    }

    // Validate deviceId format (should be 64-char hex SHA-256 hash)
    if (!/^[a-f0-9]{64}$/.test(deviceId)) {
      return new Response(JSON.stringify({ error: 'Invalid deviceId format' }), {
        status: 400,
        headers: { 'Content-Type': 'application/json' }
      });
    }

    // Retrieve encrypted blob from Netlify Blobs
    const store = getStore('sync-data');
    const encryptedBlob = await store.get(deviceId, { type: 'text' });

    if (!encryptedBlob) {
      return new Response(JSON.stringify({
        found: false,
        message: 'No sync data found for this device'
      }), {
        status: 200,
        headers: { 'Content-Type': 'application/json' }
      });
    }

    return new Response(JSON.stringify({
      found: true,
      encryptedBlob,
      timestamp: new Date().toISOString()
    }), {
      status: 200,
      headers: { 'Content-Type': 'application/json' }
    });
  } catch (error: any) {
    console.error('sync-pull error:', error);
    return new Response(JSON.stringify({
      error: 'Internal server error'
    }), {
      status: 500,
      headers: { 'Content-Type': 'application/json' }
    });
  }
};
```

Key design decisions:
- Returns 200 with `found: false` when no data exists (not 404) so client can distinguish "no sync data" from actual errors
- Validates deviceId format (64-char hex) to prevent abuse
- Uses `{ type: 'text' }` in getStore.get() to retrieve as string (not ReadableStream)
- Consistent JSON error responses across all error paths
