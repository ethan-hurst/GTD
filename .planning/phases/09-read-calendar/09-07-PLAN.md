```typescript
onMount(async () => {
  await calendarState.loadEvents();

  // If Outlook is connected, load sync meta and auto-discover calendars
  if (authState.isAuthenticated) {
    await outlookSyncState.loadSyncMeta();
    // If no calendars discovered yet, discover them
    if (outlookSyncState.calendars.length === 0) {
      await outlookSyncState.discoverCalendars();
    }
  }

  // Default to Day view on mobile (only on first load)
  if (mobileState.isMobile && !hasInitializedView) {
    calendarState.setView('timeGridDay');
    hasInitializedView = true;
  }
});
```

Update the displayEvents derived to filter Outlook events by enabled calendars:
```typescript
const displayEvents = $derived.by(() => {
  const range = visibleRange;
  let events = expandAllRecurrences(calendarState.events, range.start, range.end);

  // Filter Outlook events by enabled calendars (if any calendars are configured)
  if (outlookSyncState.calendars.length > 0) {
    const enabledCalendarIds = new Set(
      outlookSyncState.calendars
        .filter(c => c.enabled)
        .map(c => c.calendarId)
    );

    events = events.filter(e => {
      // Non-Outlook events always pass through
      if (e.syncSource !== 'outlook') return true;
      // Outlook events: check if their calendar is enabled
      return e.outlookCalendarId ? enabledCalendarIds.has(e.outlookCalendarId) : true;
    });
  }

  return events;
});
```

Update handleEventClick to show read-only info for Outlook events:
```typescript
function handleEventClick(event: CalendarEvent) {
  if (event.syncSource === 'outlook') {
    // For Outlook events, show details but don't open edit form
    // Use toast to indicate read-only
    toast.success(`${event.title}${event.location ? ' — ' + event.location : ''}`, {
      duration: 3000
    });
    return;
  }
  editingEvent = event;
  initialDate = undefined;
  showForm = true;
}
```
