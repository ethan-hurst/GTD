```svelte

    <Sidebar />

    {#if !mobileState.isMobile}
      <header class="...existing search header...">
        <SearchBar bind:this={searchBarRef} />
      </header>
    {/if}

    <main class="flex-1 overflow-y-auto pb-6 {mobileState.isMobile ? 'pb-20' : ''} bg-white dark:bg-gray-950">
      {@render children()}
    </main>

    {#if !mobileState.isMobile}
      <StatusBar />
    {/if}
  </div>
</div>

{#if mobileState.isMobile}

{/if}
```
