**Pattern for component tests in Vitest Browser Mode:**
```typescript
import { render } from 'vitest-browser-svelte'
import { expect, test } from 'vitest'
import MyComponent from './MyComponent.svelte'

test('renders correctly', async () => {
  const screen = render(MyComponent, { props: { ... } })
  // Use screen.getByRole, screen.getByText, screen.getByPlaceholder etc.
  // Use await expect.element(el).toBeInTheDocument() for assertions
  // Use await el.click() for interactions
})
```
